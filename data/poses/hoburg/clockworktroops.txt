#load helmet /data/items/races/hoburg/clockwork/helmet_clockwork_medium.txt
#load helmet /data/items/races/hoburg/normal/helmet.txt -1 -16
#load helmet /data/items/races/abysian/abysian_helmet.txt 1 -2

#load weapon /data/items/races/human/normal/humanweapon.txt 1 -5
#load weapon /data/items/races/abysian/weapon.txt 1 -5
#load weapon /data/items/races/hoburg/clockwork/weapon_clockwork_medium.txt
#load specialweapon /data/items/races/hoburg/clockwork/specialweapon_clockwork_medium.txt

#load offhand /data/items/races/human/normal/humanoffhand.txt 0 -5
#load offhand /data/items/races/abysian/offhand.txt 0 -5
#load offhand /data/items/races/human/normal/shields.txt 0 -5
#load offhand /data/items/races/human/normal/towershields.txt 0 -5

#load headslot /data/items/races/hoburg/clockwork/headslots_clockwork_medium.txt
#load handslots /data/items/races/hoburg/clockwork/handslots_clockwork_medium.txt
#load specialslot /data/items/races/hoburg/clockwork/specialslots_clockwork_medium.txt

-- slots generated here are given through items; this pattern is used to be able to generate
-- a variable number of slots; i.e. an integrated helmet or a replaceable helmet
#generateitem 1 headslot

-- these will generate some special items; some of them weapons like firespewers, some miscellaneous like supply bonuses
#additionalweaponslot 1 specialslot

#endpose
